#第十二章笔记：文件操作
1. fopen()函数实现文件的打开功能
	FILE *fp;									#定义一个文件指针
	fp=fopen("FileName","使用方式");																
	fclose(fp)									#关闭文件,fopen(),fclose()函数都在stdio.h头文件中.
2. fputc(ch,fp)										#写字符函数
3. fgetc(fp)										#读字符函数
4. feof(fp)										#!feof(fp) 判断文件是否真正结束，适用于任何文件
5. fputs(字符串,文件指针)								#写字符串函数
6. fgets(字符数组名，n, 文件指针);							#n表示所得到的字符串中字符的个数，包括"\0".
7. fprintf(文件类型指针，格式字符串，输出列表);						#格式化写入函数，将“输出列表”数据按照指定的格式输出到“文件指针名”所指向的磁盘文件。类似的是fscanf(). 
8. fread()和fwrite()数据块读写函数.

定位文件
1. fseek(文件指针类型， 位移量，起始点)							#位移量(移动的字节数)是long数据类型，以便文件大于64KB时不会出差错.当用常量表示位移量是，后面加L;起始点：0 文件首;1
											# 当前位置; 2 文件尾. 一般用于二进制文件.
2. rewind(文件类型指针)									#位置指针重新返回到文件开头
3. ftell(文件类型指针)									#得到流式文件中的当前位置，相对于文件开头的位移量来表示。返回-1L时，出错.
